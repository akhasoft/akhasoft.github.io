    Core.ConfigurationChangedListener

    public class LocationCallbacks
extends BaseActivityLifecycleProceed.BaseActivityCallbacks
implements Core.ConfigurationChangedListener
    Extends the BaseActivityLifecycleProceed.BaseActivityCallbacks class to provide location APIs support. For example, in Activity: 

     import akha.yakhont.callback.annotation.CallbacksInherited;
 import akha.yakhont.location.LocationCallbacks;
 import akha.yakhont.location.LocationCallbacks.LocationListener;

 import android.location.Location;

 @CallbacksInherited(LocationCallbacks.class)
 public class MyActivity extends Activity implements LocationListener {

     @Override
     public void onLocationChanged(Location location, Date date) {
         // your code here
     }

     public Location getLocation() {
         return LocationCallbacks.getLocationCallbacks(this).getCurrentLocation();
     }
 }
